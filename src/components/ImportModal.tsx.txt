ImportModal.tsx

import React, { useState } from 'react';
import { parseMenuFromText } from '../services/geminiService';
import { MenuItem } from '../types';

interface ImportModalProps { isOpen: boolean; onClose: () => void; onImport: (items: MenuItem[]) => void; }

const ImportModal: React.FC<ImportModalProps> = ({ isOpen, onClose, onImport }) => {
  const [text, setText] = useState('');
  const [isLoading, setIsLoading] = useState(false);

  if (!isOpen) return null;
  const handleProcess = async () => {
    setIsLoading(true);
    try { const items = await parseMenuFromText(text); onImport(items); onClose(); setText(''); } catch (e) { alert("Error"); } finally { setIsLoading(false); }
  };

  return (
    <div className="fixed inset-0 z-[70] flex items-center justify-center p-4 bg-black/60 backdrop-blur-sm">
      <div className="bg-white w-full max-w-lg rounded-xl shadow-2xl flex flex-col max-h-[90vh]">
        <div className="bg-quercus-800 text-white p-4"><h3 className="font-bold">Importar Texto</h3></div>
        <div className="p-6"><textarea className="w-full h-64 border p-2" value={text} onChange={(e) => setText(e.target.value)} placeholder="Pega aquí el texto de tu menú..."></textarea></div>
        <div className="p-4 border-t flex justify-end gap-2"><button onClick={onClose} className="px-4 py-2">Cancelar</button><button onClick={handleProcess} disabled={isLoading} className="bg-quercus-800 text-white px-4 py-2 rounded">{isLoading ? '...' : 'Importar'}</button></div>
      </div>
    </div>
  );
};
export default ImportModal;
