import React, { useState, useEffect, useRef } from 'react';
import { getSommelierRecommendation } from '../services/geminiService';
import { ChatMessage, MenuItem, Language } from '../types';

interface SommelierModalProps { isOpen: boolean; onClose: () => void; menuItems: MenuItem[]; language: Language; initialQuery?: string; }

const SommelierModal: React.FC<SommelierModalProps> = ({ isOpen, onClose, menuItems, language, initialQuery }) => {
  const [messages, setMessages] = useState<ChatMessage[]>([]);
  const [input, setInput] = useState('');
  const hasSent = useRef(false);

  useEffect(() => { if (isOpen && messages.length === 0) setMessages([{ role: 'model', text: "Hola, soy tu Sommelier." }]); }, [isOpen]);
  useEffect(() => { if (isOpen && initialQuery && !hasSent.current) { hasSent.current = true; handleSend(initialQuery); } if (!isOpen) hasSent.current = false; }, [isOpen, initialQuery]);

  const handleSend = async (txt?: string) => {
    const q = txt || input;
    if (!q.trim()) return;
    if (!txt) setInput('');
    setMessages(p => [...p, { role: 'user', text: q }]);
    const ans = await getSommelierRecommendation(q, menuItems, language);
    setMessages(p => [...p, { role: 'model', text: ans }]);
  };

  if (!isOpen) return null;
  return (
    <div className="fixed inset-0 z-50 flex items-center justify-center p-4 bg-black/50 backdrop-blur-sm">
      <div className="bg-white w-full max-w-lg h-[80vh] rounded-2xl flex flex-col">
        <div className="bg-quercus-800 text-white p-4 flex justify-between"><h2 className="font-bold">Sommelier AI</h2><button onClick={onClose}>X</button></div>
        <div className="flex-1 overflow-y-auto p-4 space-y-4 bg-gray-50">{messages.map((m, i) => (<div key={i} className={`p-3 rounded-lg max-w-[80%] ${m.role === 'user' ? 'ml-auto bg-quercus-800 text-white' : 'bg-white border'}`}>{m.text}</div>))}</div>
        <div className="p-4 border-t flex gap-2"><input className="flex-1 border p-2 rounded-full" value={input} onChange={(e) => setInput(e.target.value)} /><button onClick={() => handleSend()} className="bg-quercus-800 text-white p-2 rounded-full">Send</button></div>
      </div>
    </div>
  );
};
export default SommelierModal;
